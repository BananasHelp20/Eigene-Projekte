
#include <WiFi.h>               // ESP32 Wireless Lib
#include "ThingSpeak.h"         // Lib from Mathworks - Communication with ThingSpeak

#define LED_BUILTIN 5   // Onboard LED blue
#define DHTPIN 23       // Connection pin DHT22 (1 Wire)

#define PRINT_DEBUG_MESSAGES   

WiFiClient client;

char ssid[] = "xxxxxxxxx";                 // network SSID (name) 
char pass[] = "xxxxxxxxxx";                // network password

unsigned long ChannelNumber_System = 1234567;   //Thingspeak Channel
const char * WriteAPIKey_System = "xxxxxx";     //Thingspeak WriteAPIKey
  
float ConnTime;       // Connection Time to WLAN AP
int conCount ;        // Connection Counter
  
long startTime;  
long duration;   

void setup() 
{      
    #ifdef PRINT_DEBUG_MESSAGES    
      Serial.begin(115200);
      Serial.print("Start System...\n\n");
    #endif
  
    // initialize digital pin 
    pinMode(LED_BUILTIN, OUTPUT);
    digitalWrite(LED_BUILTIN, HIGH);   // turn the LED on (HIGH is the voltage level)

 
    //measuring connection time client -> WLAN Access Point
    startTime = millis();  
  
    //start WLAN
  if(!WiFi.begin(ssid, pass)) {
       #ifdef PRINT_DEBUG_MESSAGES
          Serial.println("Ooops, wifi begin fault");
       #endif   
    }
    
    //start Thingspeak
  if(!ThingSpeak.begin(client)) {
       #ifdef PRINT_DEBUG_MESSAGES
          Serial.println("Ooops, start Thingspeak error");
       #endif   
    }
    
    #ifdef PRINT_DEBUG_MESSAGES   
      Serial.println();
      Serial.println();
      Serial.print("Connecting to ");
      Serial.println(ssid);
    #endif
    
    conCount = 0; 
  
    // wait for WLAN Conneciton
    while (WiFi.status() != WL_CONNECTED) {
      
      digitalWrite(LED_BUILTIN, LOW);   // turn the LED off (HIGH is the voltage level)
      delay(100);
      digitalWrite(LED_BUILTIN, HIGH);   // turn the LED on (HIGH is the voltage level)
      delay(100);  
      conCount++;
  
      // timeout
      if ( conCount > 30) {
        #ifdef PRINT_DEBUG_MESSAGES  
          Serial.println("go sleeping...connection Problem to WiFi");    
        #endif
        ESP.deepSleep(120000000);
      }
     
      #ifdef PRINT_DEBUG_MESSAGES   
        Serial.print(".");
      #endif
    }
    #ifdef PRINT_DEBUG_MESSAGES
     Serial.println("");
        Serial.println("WiFi connected");
     #endif;
  
  
    duration = millis() - startTime;
    ConnTime = duration/1000.0;
    
    #ifdef PRINT_DEBUG_MESSAGES
       Serial.print("ConnectTime = "); Serial.print(ConnTime); Serial.println(" Sekunden");
    #endif
    
    long rssi = WiFi.RSSI();
  
    #ifdef PRINT_DEBUG_MESSAGES  
      Serial.print("RSSI:");
      Serial.println(rssi);
    #endif  
    
    duration = millis() - startTime;
    ConnTime = duration/1000.0;
  
    #ifdef PRINT_DEBUG_MESSAGES  
      Serial.println("ThingSpeak.writeFields start");
    #endif  
  
      ThingSpeak.setField(1,rssi);
      ThingSpeak.setField(2,"unused");
      ThingSpeak.setField(3,"unused");
      ThingSpeak.setField(4,"unused");
      ThingSpeak.setField(5,"unused");
      ThingSpeak.setField(6,"unused");
      ThingSpeak.setField(7,"unused");
      ThingSpeak.setField(8,"unused");
  
    // Write the fields that you've set all at once.
    ThingSpeak.writeFields(ChannelNumber_System, WriteAPIKey_System);  


    #ifdef PRINT_DEBUG_MESSAGES  
      Serial.println("ThingSpeak.writeFields Ende");
    #endif  
  
    #ifdef PRINT_DEBUG_MESSAGES 
      Serial.print("ConnTime = "); Serial.print(ConnTime); Serial.println(" Sekunden");
      Serial.println("go sleeping...");    
    #endif
  
    ESP.deepSleep(1000000 * 1);
  
}

void loop() {
  
  //should be never reached
  ESP.deepSleep(1000000 * 1);

}
